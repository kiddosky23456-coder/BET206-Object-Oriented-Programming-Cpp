BEGIN
    NUMBER totalVoltage = 0.0
    NUMBER voltageReading
    NUMBER averageVoltage
    INTEGER i
    
    FOR i FROM 1 TO 10 STEP 1 DO
        OUTPUT "Enter voltage reading ", i, " (V): "
        INPUT voltageReading
        totalVoltage = totalVoltage + voltageReading
    ENDFOR
    
    averageVoltage = totalVoltage / 10
    OUTPUT "Average Voltage = ", averageVoltage, " V"
END
